########################################################################
# Mongoose networking library
########################################################################
# rtl_433 vendors the amalgamated Mongoose 6.16 (src/mongoose.c +
# include/mongoose.h). Compiling that single ~16k-line file with optimization
# needs a lot of RAM, and on small machines the compiler can die building
# mongoose.c with an out-of-memory error ("virtual memory exhausted",
# "cc1: out of memory"). Enable ENABLE_COMPILE_LOW_MEMORY to re-split it into
# its ~30 original modules (via tools/unamalgam.py) so each compiles as a small
# translation unit with a much lower memory peak. This needs a Python 3
# interpreter at configure time; the default build (option OFF) does not.
# mongoose.c stays in the r_433 source list below (that list is generated by
# maintainer_update.py); when the option is ON it is excluded from the build
# and the split modules are added instead, further down.
option(ENABLE_COMPILE_LOW_MEMORY
    "Lower peak memory (RAM) to compile Mongoose on small machines by splitting mongoose.c into smaller modules; use if the compiler runs out of memory building mongoose.c (needs Python 3)"
    OFF)

if(ENABLE_COMPILE_LOW_MEMORY)
    find_program(MONGOOSE_PYTHON NAMES python3 python)
    if(NOT MONGOOSE_PYTHON)
        message(FATAL_ERROR
            "ENABLE_COMPILE_LOW_MEMORY requires a Python 3 interpreter, none found")
    endif()

    set(MONGOOSE_GEN_DIR "${CMAKE_CURRENT_BINARY_DIR}/mongoose")
    # Regenerate at configure time. Mongoose 6.16 is frozen, so this only
    # actually reruns when mongoose.c/.h or the tool change.
    execute_process(
        COMMAND "${MONGOOSE_PYTHON}"
            "${PROJECT_SOURCE_DIR}/tools/unamalgam.py"
            --amalgamated-c "${CMAKE_CURRENT_SOURCE_DIR}/mongoose.c"
            --amalgamated-h "${PROJECT_SOURCE_DIR}/include/mongoose.h"
            --out "${MONGOOSE_GEN_DIR}"
        RESULT_VARIABLE _mongoose_split_rc
        OUTPUT_VARIABLE _mongoose_split_out
        ERROR_VARIABLE _mongoose_split_out)
    if(NOT _mongoose_split_rc EQUAL 0)
        message(FATAL_ERROR "Mongoose split failed:\n${_mongoose_split_out}")
    endif()
    message(STATUS "Mongoose: split into modules under ${MONGOOSE_GEN_DIR}")

    # Fixed module list (Mongoose 6.16 is frozen), same order as the SOURCES
    # list in Cesanta's mg_modules.mk. mg_uri.c and the mg_http*.c modules
    # share file-static state, so they build as one TU (mg_http_modules.c).
    # The embedded-platform stubs (cc3200, msp432, nrf5, simplelink, lwip,
    # pic32) are generated but not compiled: they are inert on every platform
    # rtl_433 supports, and some reference headers absent from the amalgamation.
    set(MONGOOSE_SPLIT_SOURCES
        "${MONGOOSE_GEN_DIR}/common/cs_base64.c"
        "${MONGOOSE_GEN_DIR}/common/cs_dbg.c"
        "${MONGOOSE_GEN_DIR}/common/cs_dirent.c"
        "${MONGOOSE_GEN_DIR}/common/cs_time.c"
        "${MONGOOSE_GEN_DIR}/common/cs_md5.c"
        "${MONGOOSE_GEN_DIR}/common/cs_sha1.c"
        "${MONGOOSE_GEN_DIR}/common/mbuf.c"
        "${MONGOOSE_GEN_DIR}/common/mg_str.c"
        "${MONGOOSE_GEN_DIR}/common/str_util.c"
        "${MONGOOSE_GEN_DIR}/mg_net.c"
        "${MONGOOSE_GEN_DIR}/mg_net_if.c"
        "${MONGOOSE_GEN_DIR}/mg_net_if_null.c"
        "${MONGOOSE_GEN_DIR}/mg_net_if_socket.c"
        "${MONGOOSE_GEN_DIR}/mg_net_if_socks.c"
        "${MONGOOSE_GEN_DIR}/mg_ssl_if_openssl.c"
        "${MONGOOSE_GEN_DIR}/mg_ssl_if_mbedtls.c"
        "${MONGOOSE_GEN_DIR}/mg_http_modules.c"
        "${MONGOOSE_GEN_DIR}/mg_util.c"
        "${MONGOOSE_GEN_DIR}/mg_mqtt.c"
        "${MONGOOSE_GEN_DIR}/mg_mqtt_server.c"
        "${MONGOOSE_GEN_DIR}/mg_dns.c"
        "${MONGOOSE_GEN_DIR}/mg_dns_server.c"
        "${MONGOOSE_GEN_DIR}/mg_resolv.c"
        "${MONGOOSE_GEN_DIR}/mg_coap.c"
        "${MONGOOSE_GEN_DIR}/mg_sntp.c"
        "${MONGOOSE_GEN_DIR}/mg_socks.c"
        "${MONGOOSE_GEN_DIR}/common/platforms/wince/wince_libc.c"
        "${MONGOOSE_GEN_DIR}/common/platforms/windows/windows_direct.c"
    )
    # The generated umbrella mongoose.h and the sibling module headers live
    # here; put it first so it shadows the amalgamated include/mongoose.h.
    include_directories(BEFORE "${MONGOOSE_GEN_DIR}")
endif()

if("${CMAKE_C_COMPILER_ID}" STREQUAL "GNU" OR "${CMAKE_C_COMPILER_ID}" MATCHES "Clang")
    # untouched upstream code, disable all warnings
    set_source_files_properties(mongoose.c PROPERTIES COMPILE_FLAGS "-w")
endif()

if(ENABLE_COMPILE_LOW_MEMORY)
    # Exclude the amalgamation (it stays in the source list above for
    # maintainer_update.py) and build the generated split modules instead.
    set_source_files_properties(mongoose.c PROPERTIES HEADER_FILE_ONLY ON)
    target_sources(r_433 PRIVATE ${MONGOOSE_SPLIT_SOURCES})
    if("${CMAKE_C_COMPILER_ID}" STREQUAL "GNU" OR "${CMAKE_C_COMPILER_ID}" MATCHES "Clang")
        set_source_files_properties(${MONGOOSE_SPLIT_SOURCES}
            PROPERTIES COMPILE_FLAGS "-w")
    endif()
endif()

    target_link_libraries(rtl_433 "${CMAKE_THREAD_LIBS_INIT}")
